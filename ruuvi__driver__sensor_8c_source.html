157 float rd_sensor_data_parse (const rd_sensor_data_t * const provided,
158  const rd_sensor_data_fields_t requested)
159 {
160  // If there isn't valid requested data, return value "invalid".
161  if (! (provided->valid.bitfield & requested.bitfield)) { return RD_FLOAT_INVALID; }
162 
163  // If trying to get more than one field, return value "invalid".
164  if (1 != __builtin_popcount (requested.bitfield)) { return RD_FLOAT_INVALID; }
165 
166  // Return requested value
167  return provided->data[get_index_of_field (provided, requested)];
168 }

170 void rd_sensor_data_set (rd_sensor_data_t * const target,
171  const rd_sensor_data_fields_t field,
172  const float value)
173 {
174  // If there isn't valid requested data, return
175  if (! (target->fields.bitfield & field.bitfield)) { return; }
176 
177  // If trying to set more than one field, return.
178  if (1 != __builtin_popcount (field.bitfield)) { return; }
179 
180  // Set value to appropriate index
181  target->data[get_index_of_field (target, field)] = value;
182  // Mark data as valid
183  target->valid.bitfield |= field.bitfield;

186 void rd_sensor_data_populate (rd_sensor_data_t * const target,
187  const rd_sensor_data_t * const provided,
188  const rd_sensor_data_fields_t requested)
189 {
190  if (NULL == target || NULL == provided) { return; }
191 
192  // Compare provided data to requested data.
193  rd_sensor_data_fields_t available = {.bitfield = (provided->valid).bitfield & requested.bitfield};
194 
195  // We have the available, requested fields. Fill the target struct with those
196  while (available.bitfield)
197  {
198  // read rightmost field
199  rd_sensor_data_fields_t next = {.bitfield = (1 << __builtin_ctz (available.bitfield)) };
200  float value = rd_sensor_data_parse (provided, next);
201  rd_sensor_data_set (target, next, value);
202  available.bitfield &= (available.bitfield - 1); // set rightmost bit of available to 0
203  }
204 }
