164 ruuvi_driver_status_t ruuvi_interface_i2c_write_blocking (const uint8_t address,
165  uint8_t * const p_tx, const size_t tx_len, const bool stop)
166 {
167  if (!m_i2c_is_init) { return RUUVI_DRIVER_ERROR_INVALID_STATE; }
168 
169  if (NULL == p_tx) { return RUUVI_DRIVER_ERROR_NULL; }
170 
171  if (m_tx_in_progress) { return RUUVI_DRIVER_ERROR_BUSY; }
172 
173  int32_t err_code = NRF_SUCCESS;
174  m_tx_in_progress = true;
175  err_code |= nrf_drv_twi_tx (&m_twi, address, p_tx, tx_len, !stop);
176  volatile uint32_t timeout = 0;
177 
178  while (m_tx_in_progress && timeout < (timeout_us_per_byte * tx_len))
179  {
180  timeout++;

182  }
