80  uint32_t ret;

99  mp_dfu->peers_count--;

102

104  {
105  NRF_LOG_DEBUG ("Finished updating Service Changed indication for peers");
106  // We have updated Service Changed Indication for all devices.
107  ret = ble_dfu_buttonless_bootloader_start_finalize();
108 
109  if (ret != NRF_SUCCESS)

111  mp_dfu->evt_handler (BLE_DFU_EVT_BOOTLOADER_ENTER_FAILED);
112  }
113  }
114  }
115 
116  break;
117 
118  case PM_EVT_PEER_DATA_UPDATE_FAILED:
119  // Failure to update data. Service Changed cannot be sent but DFU mode is still possible
120  ret = ble_dfu_buttonless_bootloader_start_finalize();

166 static uint32_t enter_bootloader (void)
167 {
168  uint32_t ret;
169  NRF_LOG_INFO ("Writing peer data to the bootloader...");
170 
171  if (mp_dfu->is_waiting_for_svci)
172  {
173  return ble_dfu_buttonless_resp_send (DFU_OP_ENTER_BOOTLOADER, DFU_RSP_BUSY);
174  }
175 
176  // If retrieve_peer_data returns NRF_ERROR_FORBIDDEN, then the device was not bonded.
177  ret = retrieve_peer_data();
178  VERIFY_SUCCESS (ret);
179  ret = nrf_dfu_set_peer_data (&m_peer_data);
180 
181  if (ret == NRF_SUCCESS)
182  {
183  // The request was accepted. Waiting for sys events to progress.
184  mp_dfu->is_waiting_for_svci = true;
185  }
186  else if (ret == NRF_ERROR_FORBIDDEN)
187  {
188  NRF_LOG_ERROR ("The bootloader has write protected its settings page. This prohibits setting the peer data. "\
189  "The bootloader must be compiled with NRF_BL_SETTINGS_PAGE_PROTECT=0 to allow setting the peer data.");
190  }
191 
192  return ret;
193 }

227 void ble_dfu_buttonless_on_sys_evt (uint32_t sys_evt, void * p_context)
228 {
229  uint32_t ret;
230 
231  if (!nrf_dfu_set_peer_data_is_initialized())
232  {
233  return;
234  }
235 
236  ret = nrf_dfu_set_peer_data_on_sys_evt (sys_evt);
237 
238  if (ret == NRF_ERROR_INVALID_STATE)
239  {
240  // The system event is not from an operation started by buttonless DFU.
241  // No action is taken, and nothing is reported.
242  }
243  else if (ret == NRF_SUCCESS)
244  {
245  // Peer data was successfully forwarded to the Secure DFU bootloader.
246  // Set the flag indicating that we are waiting for indication response
247  // to activate the reset.
248  mp_dfu->is_waiting_for_reset = true;
249  mp_dfu->is_waiting_for_svci = false;
250  // Report back the positive response
251  ret = ble_dfu_buttonless_resp_send (DFU_OP_ENTER_BOOTLOADER, DFU_RSP_SUCCESS);
252 
253  if (ret != NRF_SUCCESS)
254  {
255  mp_dfu->evt_handler (BLE_DFU_EVT_RESPONSE_SEND_ERROR);
256  mp_dfu->is_waiting_for_reset = false;
257  }
258  }
259  else
260  {
261  // Failed to set peer data. Report this.
262  mp_dfu->is_waiting_for_reset = false;
263  mp_dfu->is_waiting_for_svci = false;
264  ret = ble_dfu_buttonless_resp_send (DFU_OP_ENTER_BOOTLOADER, DFU_RSP_BUSY);
265 
266  // Report the failure to send the response to the client
267  if (ret != NRF_SUCCESS)
268  {
269  mp_dfu->evt_handler (BLE_DFU_EVT_RESPONSE_SEND_ERROR);
270  }
271 
272  // Report the failure to enter DFU mode
273  mp_dfu->evt_handler (BLE_DFU_EVT_BOOTLOADER_ENTER_FAILED);
274  }

298 {
299  uint32_t ret;

301

304  {
305  case DFU_OP_ENTER_BOOTLOADER:
306  ret = enter_bootloader();
307 
308  if (ret == NRF_SUCCESS)

311  }
312  else if (ret == NRF_ERROR_BUSY)
313  {
314  rsp_code = DFU_RSP_BUSY;
315  }
316  else if (ret == NRF_ERROR_FORBIDDEN)
317  {
318  rsp_code = DFU_RSP_NOT_BONDED;
319  }
320 
321  break;
322 
323  default:
324  rsp_code = DFU_RSP_OP_CODE_NOT_SUPPORTED;
325  break;
326  }
327 
328  // Report back in case of error
329  if (rsp_code != DFU_RSP_SUCCESS)
330  {
331  ret = ble_dfu_buttonless_resp_send ( (ble_dfu_buttonless_op_code_t) p_evt_write->data[0],
332  rsp_code);
333 
334  if (ret != NRF_SUCCESS)
335  {
336  mp_dfu->evt_handler (BLE_DFU_EVT_RESPONSE_SEND_ERROR);
337  }
338 
339  // Report the error to the main application
340  mp_dfu->evt_handler (BLE_DFU_EVT_BOOTLOADER_ENTER_FAILED);
341  }
