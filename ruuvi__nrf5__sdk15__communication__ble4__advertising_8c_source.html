302 // Set manufacturer specific data to advertise. Clears previous data.
303 rd_status_t ri_adv_data_set (
304  const uint8_t * data, const uint8_t data_length)
305 {
306  if (NULL == data) { return RD_ERROR_NULL; }
307 
308  if (24 < data_length) { return RD_ERROR_INVALID_LENGTH; }
309 
310  // Build specification for data into ble_advdata_t advdata
311  ble_advdata_t advdata = {0};
312  // Only valid flag
313  uint8_t flags = BLE_GAP_ADV_FLAG_BR_EDR_NOT_SUPPORTED;
314  // Build manufacturer specific data
315  ble_advdata_manuf_data_t manuf_specific_data;
316  ret_code_t err_code = NRF_SUCCESS;
317  // Preserve const of data passed to us.
318  uint8_t manufacturer_data[24];
319  memcpy (manufacturer_data, data, data_length);
320  manuf_specific_data.data.p_data = manufacturer_data;
321  manuf_specific_data.data.size = data_length;
322  manuf_specific_data.company_identifier = m_adv_state.manufacturer_id;
323  // Point to manufacturer data and flags set earlier
324  advdata.flags = flags;
325  advdata.p_manuf_specific_data = &manuf_specific_data;
326 
327  // If manufacturer data is not set, assign "UNKNOWN"
328  if (0 == m_adv_state.manufacturer_id) { manuf_specific_data.company_identifier = 0xFFFF; }
329 
330  // Same buffer must not be passed to the SD on data update.
331  ble_gap_adv_data_t * p_adv_data = &m_adv_data;
332  uint8_t * p_advertisement = (advertisement_odd) ? m_advertisement0 :
333  m_advertisement1;
334  uint16_t * p_adv_len = (advertisement_odd) ? &m_adv0_len : &m_adv1_len;
335  uint8_t * p_scan = (advertisement_odd) ? m_scan0 :
336  m_scan1;
337  uint16_t * p_scan_len = (advertisement_odd) ? &m_scan0_len : &m_scan1_len;
338  m_adv0_len = sizeof (m_advertisement0);
339  m_adv1_len = sizeof (m_advertisement1);
340  advertisement_odd = !advertisement_odd;
341  // Encode data
342  err_code |= ble_advdata_encode (&advdata, p_advertisement, p_adv_len);
343  p_adv_data->adv_data.p_data = p_advertisement;
344  p_adv_data->adv_data.len = *p_adv_len;
345 
346  if (m_scannable)
347  {
348  p_adv_data->scan_rsp_data.p_data = p_scan;
349  p_adv_data->scan_rsp_data.len = *p_scan_len;
350  }
351  else
352  {
353  p_adv_data->scan_rsp_data.p_data = NULL;
354  p_adv_data->scan_rsp_data.len = 0;
355  }
356 
357  if (m_advertising)
358  {
359  err_code |= sd_ble_gap_adv_set_configure (&m_adv_handle, p_adv_data, NULL);
360  }
361 
362  return ruuvi_nrf5_sdk15_to_ruuvi_error (err_code);
363 }
