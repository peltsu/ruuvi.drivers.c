605  }

800 rd_status_t ri_lis2dh12_data_get (rd_sensor_data_t * const
801  data)
802 {
803  if (NULL == data) { return RD_ERROR_NULL; }
804 
805  rd_status_t err_code = RD_SUCCESS;
806  axis3bit16_t raw_acceleration;
807  uint8_t raw_temperature[2];
808  memset (raw_acceleration.u8bit, 0x00, 3 * sizeof (int16_t));
809  err_code |= lis2dh12_acceleration_raw_get (& (dev.ctx), raw_acceleration.u8bit);
810  err_code |= lis2dh12_temperature_raw_get (& (dev.ctx), raw_temperature);
811  // Compensate data with resolution, scale
812  float acceleration[3];
813  float temperature;
814  err_code |= rawToMg (&raw_acceleration, acceleration);
815  err_code |= rawToC (raw_temperature, &temperature);
816  uint8_t mode;
817  err_code |= ri_lis2dh12_mode_get (&mode);
818 
819  if (RD_SENSOR_CFG_SLEEP == mode) { data->timestamp_ms = dev.tsample; }
820  else if (RD_SENSOR_CFG_CONTINUOUS == mode) { data->timestamp_ms = rd_sensor_timestamp_get(); }
821  else { RD_ERROR_CHECK (RD_ERROR_INTERNAL, ~RD_ERROR_FATAL); }
822 
823  // If we have valid data, return it.
824  if (RD_UINT64_INVALID != data->timestamp_ms
825  && RD_SUCCESS == err_code)
826  {
827  rd_sensor_data_t d_acceleration;
828  float values[4];
829  rd_sensor_data_fields_t acc_fields = {.bitfield = 0};
830  d_acceleration.data = values;
831  acc_fields.datas.acceleration_x_g = 1;
832  acc_fields.datas.acceleration_y_g = 1;
833  acc_fields.datas.acceleration_z_g = 1;
834  acc_fields.datas.temperature_c = 1;
835  //Convert mG to G.
836  values[0] = acceleration[0] / 1000.0;
837  values[1] = acceleration[1] / 1000.0;
838  values[2] = acceleration[2] / 1000.0;
839  values[3] = temperature;
840  d_acceleration.valid = acc_fields;
841  d_acceleration.fields = acc_fields;
842  rd_sensor_data_populate (data,
843  &d_acceleration,
844  data->fields);
845  }
846 
847  return err_code;
848 }

949 rd_status_t ri_lis2dh12_activity_interrupt_use (const bool enable,
950  float * limit_g)
951 {
952  if (NULL == limit_g) { return RD_ERROR_NULL; }
953 
954  if (0 > *limit_g) { return RD_ERROR_INVALID_PARAM; }
955 
956  rd_status_t err_code = RD_SUCCESS;
957  lis2dh12_hp_t high_pass = LIS2DH12_ON_INT1_GEN;
958  lis2dh12_ctrl_reg6_t ctrl6 = { 0 };
959  lis2dh12_int1_cfg_t cfg = { 0 };
960  ctrl6.i2_ia1 = PROPERTY_ENABLE;
961 
962  if (enable)
963  {
964  cfg.xhie = PROPERTY_ENABLE;
965  cfg.yhie = PROPERTY_ENABLE;
966  cfg.zhie = PROPERTY_ENABLE;
967  }
968 
969  /*
970  Do not enable lower threshold on activity detection, as it would
971  turn logic into not-active detection.
972  cfg.xlie = PROPERTY_ENABLE;
973  cfg.ylie = PROPERTY_ENABLE;
974  cfg.zlie = PROPERTY_ENABLE;
975  */
976  // Adjust for scale
977  // 1 LSb = 16 mg @ FS = 2 g
978  // 1 LSb = 32 mg @ FS = 4 g
979  // 1 LSb = 62 mg @ FS = 8 g
980  // 1 LSb = 186 mg @ FS = 16 g
981  uint8_t scale;
982  uint32_t threshold;
983  float divisor;
984  err_code |= ri_lis2dh12_scale_get (&scale);
985 
986  switch (scale)
987  {
988  case 2:
989  divisor = 0.016f;
990  break;
991 
992  case 4:
993  divisor = 0.032f;
994  break;
995 
996  case 8:
997  divisor = 0.062f;
998  break;
999 
1000  case 16:
1001  divisor = 0.186f;
1002  break;
1003 
1004  default:
1005  divisor = 0.016f;
1006  break;
1007  }
1008 
1009  threshold = (uint32_t) (*limit_g / divisor) + 1;
1010 
1011  if (threshold > 0x7F) { return RD_ERROR_INVALID_PARAM; }
1012 
1013  *limit_g = threshold * divisor;
1014  // Configure highpass on INTERRUPT 1
1015  err_code |= lis2dh12_high_pass_int_conf_set (& (dev.ctx), high_pass);
1016  // Configure INTERRUPT 1 Threshold
1017  err_code |= lis2dh12_int1_gen_threshold_set (& (dev.ctx), threshold);
1018  // Configure INTERRUPT 1 ON ZHI, ZLO, YHI, YLO, XHI, XLO
1019  err_code |= lis2dh12_int1_gen_conf_set (& (dev.ctx), &cfg);
1020  // Route INTERRUPT 1 to PIN 2
1021  err_code |= lis2dh12_pin_int2_config_set (& (dev.ctx), &ctrl6);
1022  return err_code;
1023 }
