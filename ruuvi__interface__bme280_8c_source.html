410 ruuvi_driver_status_t ruuvi_interface_bme280_dsp_get (uint8_t * dsp, uint8_t * parameter)
411 {
412  if (NULL == dsp || NULL == parameter) { return RUUVI_DRIVER_ERROR_NULL; }
413 
414  ruuvi_driver_status_t err_code = RUUVI_DRIVER_SUCCESS;
415  err_code |= BME_TO_RUUVI_ERROR (bme280_get_sensor_settings (&dev));
416 
417  if (RUUVI_DRIVER_SUCCESS != err_code) { return err_code; }
418 
419  // Assume default / 0,
420  *dsp = RUUVI_DRIVER_SENSOR_CFG_DEFAULT;
421  *parameter = RUUVI_DRIVER_SENSOR_CFG_DEFAULT;
422 
423  // Check if IIR has been set. If yes, read DSP param from there.
424  if (BME280_FILTER_COEFF_OFF != dev.settings.filter)
425  {
426  *dsp |= RUUVI_DRIVER_SENSOR_DSP_LOW_PASS;
427 
428  switch (dev.settings.filter)
429  {
430  case BME280_FILTER_COEFF_2:
431  *parameter = 2;
432  break;
433 
434  case BME280_FILTER_COEFF_4:
435  *parameter = 4;
436  break;
437 
438  case BME280_FILTER_COEFF_8:
439  *parameter = 8;
440  break;
441 
442  case BME280_FILTER_COEFF_16:
443  *parameter = 16;
444  break;
445  }
446  }
447 
448  // Check if OS has been set. If yes, read DSP param from there.
449  // Param should be same for OS and IIR if it is >1.
450  // OSR is same for every element.
451  if (BME280_NO_OVERSAMPLING != dev.settings.osr_h
452  && BME280_OVERSAMPLING_1X != dev.settings.osr_h)
453  {
454  *dsp |= RUUVI_DRIVER_SENSOR_DSP_OS;
455 
456  switch (dev.settings.osr_h)
457  {
458  case BME280_OVERSAMPLING_2X:
459  *parameter = 2;
460  break;
461 
462  case BME280_OVERSAMPLING_4X:
463  *parameter = 4;
464  break;
465 
466  case BME280_OVERSAMPLING_8X:
467  *parameter = 8;
468  break;
469 
470  case BME280_OVERSAMPLING_16X:
471  *parameter = 16;
472  break;
473  }
474  }
475 
476  return RUUVI_DRIVER_SUCCESS;
477 }

479 ruuvi_driver_status_t ruuvi_interface_bme280_mode_set (uint8_t * mode)
480 {
481  if (NULL == mode) { return RUUVI_DRIVER_ERROR_NULL; }
482 
483  ruuvi_driver_status_t err_code = RUUVI_DRIVER_SUCCESS;
484  uint8_t current_mode = RUUVI_DRIVER_SENSOR_CFG_SLEEP;
485 
486  switch (*mode)
487  {
488  case RUUVI_DRIVER_SENSOR_CFG_SLEEP:
489  err_code = BME_TO_RUUVI_ERROR (bme280_set_sensor_mode (BME280_SLEEP_MODE, &dev));
490  break;
491 
492  case RUUVI_DRIVER_SENSOR_CFG_SINGLE:
493  // Do nothing if sensor is in continuous mode
494  ruuvi_interface_bme280_mode_get (&current_mode);
495 
496  if (RUUVI_DRIVER_SENSOR_CFG_CONTINUOUS == current_mode)
497  {
498  *mode = RUUVI_DRIVER_SENSOR_CFG_CONTINUOUS;
499  return RUUVI_DRIVER_ERROR_INVALID_STATE;
500  }
501 
502  err_code = BME_TO_RUUVI_ERROR (bme280_set_sensor_mode (BME280_FORCED_MODE, &dev));
503  // We assume that dev struct is in sync with the state of the BME280 and underlying interface
504  // which has the number of settings as 2^OSR is not changed.
505  // We also assume that each element runs same OSR
506  uint8_t samples = 1 << (dev.settings.osr_h - 1);
507  ruuvi_interface_delay_ms (bme280_max_meas_time (samples));
508  tsample = ruuvi_driver_sensor_timestamp_get();
509  // BME280 returns to SLEEP after forced sample
510  *mode = RUUVI_DRIVER_SENSOR_CFG_SLEEP;
511  break;
512 
513  case RUUVI_DRIVER_SENSOR_CFG_CONTINUOUS:
514  err_code = BME_TO_RUUVI_ERROR (bme280_set_sensor_mode (BME280_NORMAL_MODE, &dev));
515  break;
516 
517  default:
518  err_code = RUUVI_DRIVER_ERROR_INVALID_PARAM;
519  break;
520  }
521 
522  return err_code;
523 }
