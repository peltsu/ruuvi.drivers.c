114 ruuvi_driver_status_t ruuvi_interface_communication_nfc_init (
115  ruuvi_interface_communication_t * const channel)
116 {
117  if (NULL == channel) { return RUUVI_DRIVER_ERROR_NULL; }
118 
119  if (nrf5_sdk15_nfc_state.initialized) { return RUUVI_DRIVER_ERROR_INVALID_STATE; }
120 
121  // Set up NFC
122  ret_code_t err_code = NRF_SUCCESS;
123  err_code |= nfc_t4t_setup (nfc_callback, NULL);
124  memset (nrf5_sdk15_nfc_state.nfc_ndef_msg, 0, sizeof (nrf5_sdk15_nfc_state.nfc_ndef_msg));
125  // Run Read-Write mode for Type 4 Tag platform
126  err_code |= nfc_t4t_ndef_rwpayload_set (nrf5_sdk15_nfc_state.nfc_ndef_msg,
127  sizeof (nrf5_sdk15_nfc_state.nfc_ndef_msg));
128  // Start sensing NFC field
129  err_code |= nfc_t4t_emulation_start();
130  nrf5_sdk15_nfc_state.initialized = true;
131  nrf5_sdk15_nfc_state.msg_index = 0;
132  // Setup communication abstraction fps
133  channel->init = ruuvi_interface_communication_nfc_init;
134  channel->uninit = ruuvi_interface_communication_nfc_uninit;
135  channel->read = ruuvi_interface_communication_nfc_receive;
136  channel->send = ruuvi_interface_communication_nfc_send;
137  channel->on_evt = NULL;
138  nrf5_sdk15_nfc_state.on_nfc_evt = & (channel->on_evt);
139  return ruuvi_nrf5_sdk15_to_ruuvi_error (err_code);
140 }

285 /* Read and parse RX buffer into records. Parse records into Ruuvi Communication messages.
286  * Restore original data after last record has been parsed
287  *
288  * parameter msg: Ruuvi Communication message, received record payload is copied into message payload field.
289  *
290  * Return RUUVI_DRIVER_STATUS_MORE_AVAILABLE if payload was parsed into msg and more data is available
291  * Return RUUVI_SUCCESS if payload was parsed into msg and no more data is available
292  * Return RUUVI_DRIVER_ERROR_NOT_FOUND if no data was buffered and message could not be parsed.
293  * Return RUUVI_ERROR_DATA_SIZE if received message could not fit into message payload
294  */
295 ruuvi_driver_status_t ruuvi_interface_communication_nfc_receive (
296  ruuvi_interface_communication_message_t * msg)
297 {
298  // Input check
299  //ruuvi_platform_log("Getting message, state check");
300  if (NULL == msg) { return RUUVI_DRIVER_ERROR_NULL; }
301 
302  // State check. Do not process data while connection is active, i.e. while
303  // data might be received.
304  // if ( nrf5_sdk15_nfc_state.connected) { return RUUVI_DRIVER_ERROR_INVALID_STATE; }
305  // If new data is not received, return not found
306  if (!nrf5_sdk15_nfc_state.rx_updated) { return RUUVI_DRIVER_ERROR_NOT_FOUND; }
307 
308  ruuvi_driver_status_t err_code = RUUVI_DRIVER_SUCCESS;
309 
310  // If we're at index 0, parse message into records
311  if (0 == nrf5_sdk15_nfc_state.msg_index)
312  {
313  uint32_t desc_buf_len = sizeof (nrf5_sdk15_nfc_state.desc_buf);
314  uint32_t data_lenu32 = sizeof (
315  nrf5_sdk15_nfc_state.nfc_ndef_msg); // Skip NFCT4T length bytes?
316  err_code = ndef_msg_parser (nrf5_sdk15_nfc_state.desc_buf,
317  &desc_buf_len,
318  nrf5_sdk15_nfc_state.nfc_ndef_msg + 2, // Skip NFCT4T length bytes
319  &data_lenu32);
320  // PLATFORM_LOG_INFO("Found %d messages", ((nfc_ndef_msg_desc_t*)desc_buf)->record_count);
321  // ndef_msg_printout((nfc_ndef_msg_desc_t*) nrf5_sdk15_nfc_state.desc_buf);
322  }
323 
324  // If there is a new message, parse the payload into Ruuvi Message.
325  if (nrf5_sdk15_nfc_state.msg_index < ( (nfc_ndef_msg_desc_t *)
326  nrf5_sdk15_nfc_state.desc_buf)->record_count)
327  {
328  // PLATFORM_LOG_INFO("Parsing message %d", msg_index);
329  nfc_ndef_record_desc_t * const p_rec_desc = ( (nfc_ndef_msg_desc_t *)
330  nrf5_sdk15_nfc_state.desc_buf)->pp_record[nrf5_sdk15_nfc_state.msg_index];
331  nfc_ndef_bin_payload_desc_t * p_bin_pay_desc = p_rec_desc->p_payload_descriptor;
332 
333  // Data length check
334  if (p_bin_pay_desc->payload_length > msg->data_length) { err_code = RUUVI_DRIVER_ERROR_DATA_SIZE; }
335  else
336  {
337  memcpy (msg->data, (uint8_t *) p_bin_pay_desc->p_payload, p_bin_pay_desc->payload_length);
338  msg->data_length = p_bin_pay_desc->payload_length;
339  }
340 
341  nrf5_sdk15_nfc_state.msg_index++;
342 
343  if (RUUVI_DRIVER_SUCCESS == err_code) { err_code = RUUVI_DRIVER_STATUS_MORE_AVAILABLE; }
344  }
345 
346  // If no more records could/can be parsed, reset buffer and message counter
347  if (RUUVI_DRIVER_STATUS_MORE_AVAILABLE != err_code
348  || nrf5_sdk15_nfc_state.msg_index == ( (nfc_ndef_msg_desc_t *)
349  nrf5_sdk15_nfc_state.desc_buf)->record_count)
350  {
351  nrf5_sdk15_nfc_state.msg_index = 0;
352  nrf5_sdk15_nfc_state.rx_updated = false;
353 
354  // If tag is not writeable, restore original data
355  if (nrf5_sdk15_nfc_state.configurable)
356  {
357  ruuvi_interface_communication_nfc_data_set();
358  }
359  }
360 
361  return err_code;
362 }
