250 // Start single on command, mark autorefresh with continuous
251 ruuvi_driver_status_t ruuvi_interface_shtcx_mode_set (uint8_t * mode)
252 {
253  if (NULL == mode) { return RUUVI_DRIVER_ERROR_NULL; }
254 
255  // Enter sleep by default and by explicit sleep commmand
256  if (RUUVI_DRIVER_SENSOR_CFG_SLEEP == *mode || RUUVI_DRIVER_SENSOR_CFG_DEFAULT == *mode)
257  {
258  m_autorefresh = false;
259  *mode = RUUVI_DRIVER_SENSOR_CFG_SLEEP;
260  return RUUVI_DRIVER_SUCCESS;
261  }
262 
263  if (RUUVI_DRIVER_SENSOR_CFG_SINGLE == *mode)
264  {
265  // Do nothing if sensor is in continuous mode
266  uint8_t current_mode;
267  ruuvi_interface_shtcx_mode_get (&current_mode);
268 
269  if (RUUVI_DRIVER_SENSOR_CFG_CONTINUOUS == current_mode)
270  {
271  *mode = RUUVI_DRIVER_SENSOR_CFG_CONTINUOUS;
272  // Start first measurement
273  shtc1_measure();
274  return RUUVI_DRIVER_ERROR_INVALID_STATE;
275  }
276 
277  // Enter sleep after measurement
278  m_autorefresh = false;
279  *mode = RUUVI_DRIVER_SENSOR_CFG_SLEEP;
280  m_tsample = ruuvi_driver_sensor_timestamp_get();
281  return SHTCX_TO_RUUVI_ERROR (shtc1_measure_blocking_read (&m_temperature, &m_humidity));
282  }
283 
284  if (RUUVI_DRIVER_SENSOR_CFG_CONTINUOUS == *mode)
285  {
286  m_autorefresh = true;
287  return RUUVI_DRIVER_SUCCESS;
288  }
289 
290  return RUUVI_DRIVER_ERROR_INVALID_PARAM;
291 }

310 ruuvi_driver_status_t ruuvi_interface_shtcx_data_get (ruuvi_driver_sensor_data_t * const
311  p_data)
312 {
313  if (NULL == p_data) { return RUUVI_DRIVER_ERROR_NULL; }
314 
315  ruuvi_driver_status_t err_code = RUUVI_DRIVER_SUCCESS;
316 
317  if (m_autorefresh)
318  {
319  /* Sensor sleep clears measured values, blocking read required.
320  // read sensor values
321  err_code |= SHTCX_TO_RUUVI_ERROR(shtc1_read(&m_temperature, &m_humidity));
322  // Start next measurement
323  err_code |= SHTCX_TO_RUUVI_ERROR(shtc1_measure());
324  */
325  err_code |= SHTCX_TO_RUUVI_ERROR (shtc1_measure_blocking_read (&m_temperature,
326  &m_humidity));
327  m_tsample = ruuvi_driver_sensor_timestamp_get();
328  }
329 
330  if (RUUVI_DRIVER_SUCCESS == err_code && RUUVI_DRIVER_UINT64_INVALID != m_tsample)
331  {
332  ruuvi_driver_sensor_data_t d_environmental;
333  ruuvi_driver_sensor_data_fields_t env_fields = {.bitfield = 0};
334  float env_values[2];
335  env_values[0] = m_humidity / 1000.0f;
336  env_values[1] = m_temperature / 1000.0f;
337  env_fields.datas.humidity_rh = 1;
338  env_fields.datas.temperature_c = 1;
339  d_environmental.data = env_values;
340  d_environmental.valid = env_fields;
341  d_environmental.fields = env_fields;
342  ruuvi_driver_sensor_data_populate (p_data,
343  &d_environmental,
344  p_data->fields);
345  p_data->timestamp_ms = m_tsample;
346  }
347 
348  return err_code;
349 }
