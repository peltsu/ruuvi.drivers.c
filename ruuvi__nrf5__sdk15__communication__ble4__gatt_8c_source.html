462  } break;
463 
464  case PM_EVT_CONN_SEC_SUCCEEDED:
465  {
466  } break;
467 
468  case PM_EVT_CONN_SEC_FAILED:
469  {
470  /* Often, when securing fails, it shouldn't be restarted, for security reasons.
471  * Other times, it can be restarted directly.
472  * Sometimes it can be restarted, but only after changing some Security Parameters.
473  * Sometimes, it cannot be restarted until the link is disconnected and reconnected.

484  break;

488  // Run garbage collection on the flash.
489  err_code = fds_gc();
490 
491  if (err_code == FDS_ERR_NO_SPACE_IN_QUEUES)
492  {
493  // Retry.
494  }
495  else
496  {
497  APP_ERROR_CHECK (err_code);
498  }
499  }
500  break;
501 
502  case PM_EVT_PEERS_DELETE_SUCCEEDED:
503  {
504  } break;
505 
506  case PM_EVT_PEER_DATA_UPDATE_FAILED:
507  {
508  // Assert.
